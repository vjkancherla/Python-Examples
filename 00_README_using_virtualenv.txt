VirtualENV with Python3

1. mkdir /Documents/python_code/virtual_envs/py3
2. cd /Documents//python_code/
3. /Users/vija0326/Library/Python/2.7/bin/virtualenv -p python3 ./virtual_envs/py3
4. . ./virtual_envs/py3/bin/activate or source ./virtual_envs/py3/bin/activate

5. . ./virtual_envs/py3/bin/deactivate to exit or just "deactivate"
